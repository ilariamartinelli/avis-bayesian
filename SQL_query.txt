SELECT A.CAI, A.ID_PRES, B.PMIN, C.PMAX, A.POLSO, D.EMOGLOBINA, PRE.DTPRES
FROM  (SELECT CAI, ID_PRES, ESITO_NUMERICO AS POLSO
      FROM PARAMETRI_PRELIEVO
      WHERE (ID_DESC_PAPREL = 'POLSO')) AS A INNER JOIN
      (SELECT CAI, ID_PRES, ESITO_NUMERICO AS PMIN
      FROM PARAMETRI_PRELIEVO AS PARAMETRI_PRELIEVO_4
      WHERE (ID_DESC_PAPREL = 'PMIN')) AS B ON A.CAI = B.CAI AND A.ID_PRES = B.ID_PRES INNER JOIN
      (SELECT CAI, ID_PRES, ESITO_NUMERICO AS PMAX
      FROM PARAMETRI_PRELIEVO AS PARAMETRI_PRELIEVO_3
      WHERE (ID_DESC_PAPREL = 'PMAX')) AS C ON A.CAI = C.CAI AND A.ID_PRES = C.ID_PRES INNER JOIN
      (SELECT CAI, ID_PRES, ESITO_NUMERICO AS EMOGLOBINA
      FROM PARAMETRI_PRELIEVO AS PARAMETRI_PRELIEVO_2
      WHERE (ID_DESC_PAPREL = '613')) AS D ON A.CAI = D.CAI AND A.ID_PRES = D.ID_PRES INNER JOIN
      (SELECT CAI, DTPRES, ID_PRES
      FROM PRESENTAZIONI
      WHERE (ID_PUNTPREL = '100') AND (TIPO_ATTIVITA = 'D') AND (DATEPART(YEAR, DTPRES) >= 2010))
      AS PRE ON A.ID_PRES = PRE.ID_PRES
ORDER BY A.CAI, A.ID_PRES
